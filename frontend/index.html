Quick start

Run an attack campaign in minutes.

Install the CLI, point it at an agent endpoint, select a profile, and generate security plus governance evidence.

pip install sagsbench

sagsbench scan \
  --target http://localhost:8000/chat \
  --profile enterprise \
  --report html,json,sarif
from sagsbench import Campaign
from sagsbench.adapters import OpenAICompatibleTarget

target = OpenAICompatibleTarget(
    name="support-agent",
    base_url="https://api.openai.com/v1",
    model="gpt-4o-mini",
)

Campaign.enterprise(target=target).run().to_html("reports/sagsbench.html")
name: SAGSBench Red Team
on: [pull_request]
jobs:
  scan:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: pip install sagsbench
      - run: sagsbench scan --target http://localhost:8000/chat --report sarif
